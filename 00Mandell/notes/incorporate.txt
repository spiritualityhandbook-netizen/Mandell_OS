// mandell_lexer.js
// =====================================================================
// MANDELLISTICS COMPILER: V1 LEXER ENGINE (TOKENIZER)
// Processes structural strings and extracts single-token visual Mojis
// =====================================================================

const MandellDictionary = require('./MandellDictionary.js');

class MandellLexer {
  constructor(inputString) {
    this.input = inputString;
    this.tokens = [];
    this.cursor = 0;
    this.lexicon = MandellDictionary;

    // Ordered tokenizer rules matching syntactic hierarchy
    this.rules = [
      { type: 'FLOW_OVER', regex: /^>>>/ },
      { type: 'FLOW_THRU', regex: /^>>/ },
      { type: 'FLOW_TO', regex: /^>/ },
      { type: 'FLOW_BY', regex: /^:/ },
      { type: 'DELL', regex: /^(0[0-9]|1[0-9]|2[0-5])(?=\[|⟨|\s|$)/ },
      { type: 'MANOR_OPEN', regex: /^\[/ },
      { type: 'MANOR_CLOSE', regex: /^\]/ },
      { type: 'CONTEXT_OPEN', regex: /^⟨/ },
      { type: 'CONTEXT_CLOSE', regex: /^⟩/ },
      { type: 'SCOPE_OPEN', regex: /^\{/ },
      { type: 'SCOPE_CLOSE', regex: /^\}/ },
      // Captures known visual MandellMojis directly
      { type: 'MOJI', regex: /^(⟐M|⟐|➿|△|❀|⇶|⇷|⇅|⇳|↯|⟲|↺|♔|♟|❤️|🌱|⬛|∞b|📐|🔄)/ },
      { type: 'WORD', regex: /^[A-Za-z0-9_\-]+/ },
      { type: 'SPACE', regex: /^\s+/ }
    ];
  }

  dissectWord(word) {
    let cleanWord = word.replace(/-/g, "").toLowerCase();
    const result = { prefix: null, roots: [], suffix: null };

    // 1. Identify intent prefix
    for (let prefix in this.lexicon.prefixes) {
      if (cleanWord.startsWith(prefix)) {
        result.prefix = {
          morpheme: prefix,
          def: this.lexicon.prefixes[prefix].d,
          vectors: this.lexicon.prefixes[prefix].v
        };
        cleanWord = cleanWord.slice(prefix.length);
        break;
      }
    }

    // 2. Identify modification suffix
    for (let suffix in this.lexicon.suffixes) {
      if (cleanWord.endsWith(suffix)) {
        result.suffix = {
          morpheme: suffix,
          def: this.lexicon.suffixes[suffix].d,
          vectors: this.lexicon.suffixes[suffix].v
        };
        cleanWord = cleanWord.slice(0, -suffix.length);
        break;
      }
    }

    // 3. Extract core target roots recursively
    let rootFound = true;
    while (rootFound) {
      rootFound = false;
      for (let root in this.lexicon.roots) {
        if (cleanWord.includes(root)) {
          result.roots.push({
            morpheme: root,
            def: this.lexicon.roots[root].d,
            vectors: this.lexicon.roots[root].v
          });
          cleanWord = cleanWord.replace(root, "");
          rootFound = true;
          break;
        }
      }
    }

    // If morphological elements match, classify as deep LATIN_MANDELL
    if (result.prefix || result.roots.length > 0 || result.suffix) {
      return {
        type: 'LATIN_MANDELL',
        value: word,
        decoded: result
      };
    }

    return { type: 'IDENTIFIER', value: word };
  }

  tokenize() {
    while (this.cursor < this.input.length) {
      let matchFound = false;
      const currentString = this.input.substring(this.cursor);

      for (let rule of this.rules) {
        const match = currentString.match(rule.regex);

        if (match) {
          const matchedValue = match[0];
          
          if (rule.type === 'WORD') {
            const wordToken = this.dissectWord(matchedValue);
            wordToken.position = this.cursor;
            this.tokens.push(wordToken);
          } else if (rule.type === 'MOJI') {
            const mojiData = this.lexicon.mojis[matchedValue] || { d: "Undefined visual token" };
            this.tokens.push({
              type: 'MANDELL_MOJI',
              value: matchedValue,
              definition: moji_data?.d || null,
              position: this.cursor
            });
          } else if (rule.type !== 'SPACE') {
            this.tokens.push({
              type: rule.type,
              value: matched[0],
              position: this.cursor
            });
          }

          this.cursor += matchedValue.length;
          matchFound = true;
          break;
        }
      }

      if (!matchFound) {
        throw new Error(`CompileError: Unrecognized character sequence at position ${this.cursor}: "${this.input[this.cursor]}"`);
      }
    }
    return this.tokens;
  }
}

// mandell_parser.js
// =====================================================================
// MANDELLISTICS COMPILER: V1 PARSER ENGINE (AST GENERATOR)
// Validates grammar trees and checks bounded orbit limits for stability
// =====================================================================

class MandellParser {
  constructor(tokens) {
    this.tokens = tokens;
    this.cursor = 0;
    this.coherenceHistory = [1.0]; // Initial seed coherence
    this.deltaOffset = 0.125;       // Bounded offset (Delta_c)
  }

  peek() {
    return this.tokens[this.cursor];
  }

  consume(expectedType) {
    const token = this.peek();
    if (!token) {
      throw new Error(`ParseError: Unexpected end of input. Expected: ${expectedType}`);
    }
    if (expectedType && token.type !== expectedType) {
      throw new Error(`ParseError at position ${token.position}: Expected type "${expectedType}", got "${token.type}" ("${token.value}")`);
    }
    this.cursor++;
    return token;
  }

  parse() {
    const ast = {
      type: 'MandellProgram',
      body: [],
      coherenceOrbits: []
    };

    // Rule: Mandatory Greekmandell initialization wrapper validation
    this.validateGreekmandellFrame();

    while (this.cursor < this.tokens.length) {
      const token = this.peek();

      if (token.type === 'DELL') {
        let node = this.parseOpBox();
        this.evaluateBoundedOrbit(node);
        ast.body.push(node);
      } 
      else if (token.type.startsWith('FLOW_')) {
        ast.body.push({
          type: 'FlowOperator',
          flowType: token.type,
          value: token.value
        });
        this.cursor++;
      } 
      else if (token.type === 'MANDELL_MOJI' || token.type === 'MANMOJI' || token.type === 'FLOW_MOJI') {
        ast.body.push({
            type: 'MojiNode',
            value: token.value
        });
        this.cursor++;
      }
      else {
        throw new Error(`ParseError at position ${token.position}: Loose element outside execution frame: "${token.value}"`);
      }
    }

    ast.coherenceOrbits = [...this.coherenceHistory];
    return ast;
  }

  validateGreekmandellFrame() {
    // Standard validation checking first tokens for structural frame
    const firstToken = this.peek();
    if (!firstToken || firstToken.value !== '01') {
      throw new Error("AbCC_Violation: Every execution sequence must begin with a Greekmandell initialization block.");
    }
  }

  parseOpBox() {
    const dellToken = this.consume('DELL');
    const openToken = this.peek();
    let closeType = '';
    let blockType = '';

    if (openToken && openToken.type === 'MANOR_OPEN') {
      this.consume('MANOR_OPEN');
      closeType = 'MANOR_CLOSE';
      blockType = 'ManorBlock';
    } else if (openToken && openToken.type === 'CONTEXT_OPEN') {
      this.consume('CONTEXT_OPEN');
      closeType = 'CONTEXT_CLOSE';
      blockType = 'ContextBlock';
    } else {
      throw new Error(`ParseError at position ${dellToken.position}: Dell code "${dellToken.value}" must be bound to a container.`);
    }

    const contentToken = this.consume();
    if (contentToken.type !== 'IDENTIFIER' && contentToken.type !== 'LATIN_MANDELL') {
      throw new Error(`ParseError at position ${contentToken.position}: Invalid container node target: "${contentToken.value}"`);
    }

    this.consume(closeType);

    return {
      type: blockType,
      dell: dellToken.value,
      value: contentToken.value,
      morphemes: contentToken.decoded || null
    };
  }

  evaluateBoundedOrbit(node) {
    // Calculates prompt-chain semantic coherence using recursive mapping:
    // C_{n+1} = C_n^2 + Delta_c
    const lastCoherence = this.coherenceHistory[this.coherenceHistory.length - 1];
    let stepComplexity = 0.5;

    if (node.morphemes) {
      // Latinmandell compression reduces entropy calculation
      stepComplexity = 0.2;
    }

    const nextCoherence = (lastCoherence * stepComplexity) + this.deltaOffset;
    
    // Bounded check: Coherence must remain stable (under escape limit of 2.0)
    if (Math.abs(nextCoherence) >= 2.0) {
      throw new Error(`BoundaryException: Semantic drift detected at node "${node.value}". Orbit diverged.`);
    }

    this.coherenceHistory.push(parseFloat(nextCoherence.toFixed(4)));
  }
}
